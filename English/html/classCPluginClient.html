Detailed Description

The plugin interface.

Create CConnecter instance. The interface only is implemented by plug-in.

  • blocking model: It may be used with CConnecterThread for connection is blocking model.
  • no-blocking model: CPluginClientThread with CConnecterConnect for a single background thread handles multiple remote desktop connections. (Not used yet)

CConnecter* CClient::CreateConnecter(const QString& id)
{
CConnecter* pConnecter = nullptr;
auto it = m_Plugins.find(id);
if(m_Plugins.end() != it)
{
bool bRet = 0;
qDebug(log) << "CreateConnecter id:" << id;
auto plugin = it.value();
if(plugin) {
//p = plugin->CreateConnecter(id);
bRet = QMetaObject::invokeMethod(
plugin,
"CreateConnecter",
Qt::DirectConnection,
Q_RETURN_ARG(CConnecter*, pConnecter),
Q_ARG(QString, id),
Q_ARG(CParameterClient*, m_pParameterClient));
if(!bRet) {
qCritical(log) << "Create CConnecter fail.";
return nullptr;
}
}
}
return pConnecter;
}
